(define (domain logistics)
  (:requirements
    :strips
    :typing
    :negative-preconditions
    :disjunctive-preconditions
    :equality
    :conditional-effects
  )

  (:types
    location
    city
    physobj
    package
    vehicle
    airport
  )

  (:predicates
    (loc ?l - location ?c - city)
    (at ?x - physobj ?l - location)
    (in ?p - package ?v - vehicle)
  )

  (:action load
    :parameters (?p - package ?v - vehicle ?l - location)
    :precondition (and (at ?v ?l) (at ?p ?l))
    :effect (and (in ?p ?v))
  )

  (:action unload
    :parameters (?p - package ?v - vehicle ?l - location)
    :precondition (and (in ?p ?v) (at ?v ?l))
    :effect (and (not (in ?p ?v)) (at ?p ?l))
  )

  (:action drive
    :parameters (?v - vehicle ?l1 - location ?l2 - location)
    :precondition (and (at ?v ?l1) (loc ?l1 ?c) (loc ?l2 ?c))
    :effect (and (at ?v ?l2))
  )

  (:action fly
    :parameters (?a1 - airport ?a2 - airport)
    :precondition (at ?a1 ?a1)
    :effect (at ?a2 ?a2)
  )
)
